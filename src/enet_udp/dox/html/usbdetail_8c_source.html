287 void usb_event(USBDriver *usbp, usbevent_t event) {
288 
289  switch (event) {
290  case USB_EVENT_RESET:
291  return;
292  case USB_EVENT_ADDRESS:
293  return;
294  case USB_EVENT_CONFIGURED:
295  chSysLockFromIsr();
296 
297  /* Enables the endpoints specified into the configuration.
298  Note, this callback is invoked from an ISR so I-Class functions
299  must be used.*/
300  usbInitEndpointI(usbp, USB_CDC_DATA_REQUEST_EP, &ep1config);
301  usbInitEndpointI(usbp, USB_CDC_INTERRUPT_REQUEST_EP, &ep2config);
302 
303  /* Resetting the state of the CDC subsystem.*/
304  sduConfigureHookI(usbp);
305 
306  chSysUnlockFromIsr();
307  return;
308  case USB_EVENT_SUSPEND:
309  return;
310  case USB_EVENT_WAKEUP:
311  return;
312  case USB_EVENT_STALLED:
313  return;
314  }
315  return;
316 }
